    loc_data=file.iloc[:,x+2]
    loc_column = file.columns[x+2]
    outlier_iqr =[]
    for y in loc_data:
        if ((y > max_iqr_score) or (y < min_iqr_score)):
            outlier_iqr.append(y)
    print(loc_column,outlier_iqr)
